# =========================================
# Modèle IA — Estimation Consommation Énergie
# Variables : Température + Occupation
# =========================================

import numpy as np

# ---- 1. Génération de données synthétiques ----
np.random.seed(42)
n_samples = 200

temperature = np.random.uniform(-10, 40, n_samples)   # °C
occupation  = np.random.uniform(0,   200, n_samples)  # personnes

# Consommation réelle (avec bruit) en kWh
w1_reel, w2_reel, b_reel = 2.8, 1.5, 80
bruit = np.random.normal(0, 10, n_samples)
consommation = w1_reel * temperature + w2_reel * occupation + b_reel + bruit


# ---- 2. Régression linéaire (modèle IA simple) ----
class ModeleEnergie:
    def __init__(self):
        self.w1 = 0.0
        self.w2 = 0.0
        self.b  = 0.0

    def predire(self, temp, occ):
        """Prédiction : y = w1*temp + w2*occ + b"""
        return self.w1 * temp + self.w2 * occ + self.b

    def entrainer(self, X1, X2, Y, lr=0.0001, epochs=1000):
        n = len(Y)
        for epoch in range(epochs):
            # Prédictions actuelles
            Y_pred = self.w1 * X1 + self.w2 * X2 + self.b

            # Calcul des erreurs (MSE)
            erreur = Y_pred - Y

            # Gradients (descente de gradient)
            dw1 = (2/n) * np.dot(erreur, X1)
            dw2 = (2/n) * np.dot(erreur, X2)
            db  = (2/n) * np.sum(erreur)

            # Mise à jour des poids
            self.w1 -= lr * dw1
            self.w2 -= lr * dw2
            self.b  -= lr * db

            if epoch % 100 == 0:
                mse = np.mean(erreur**2)
                print(f"Époque {epoch:4d} | MSE = {mse:.2f}")


# ---- 3. Entraînement ----
modele = ModeleEnergie()
modele.entrainer(temperature, occupation, consommation, lr=0.0001, epochs=1000)

print(f"\nPoids appris : w1={modele.w1:.3f}, w2={modele.w2:.3f}, b={modele.b:.3f}")


# ---- 4. Prédiction sur un nouveau bâtiment ----
temp_test = 15    # °C
occ_test  = 50    # personnes

prediction = modele.predire(temp_test, occ_test)
print(f"\n→ Consommation estimée : {prediction:.1f} kWh")
print(f"  (Valeur réelle approx.: {w1_reel*temp_test + w2_reel*occ_test + b_reel:.1f} kWh)")
